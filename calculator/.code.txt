 char actions[] = {'+', '-', '*', '/', '\0'};
    char lastAction = '+';
 {
        char operation[] = "11+22+33-66+10*3/10";
        printf("Dzialanie: %s\n", operation);

        int i = 0, cellIndex = 0;
        float a = 0, b = 0;
        bool isASet = false, isBSet = false;

        char cell[] = {'\0', '\0', '\0', '\0'};

        int length = sizeof(operation) / sizeof(operation[0]);
        for (i = 0; i < length; i++)
        {
            if (!(arrayContain(actions, operation[i])) && operation[i] != '\0')
            {
                cell[cellIndex] = operation[i];
                cellIndex++;
                continue;
            }

            cellIndex = 0;

            if (!isASet)
            {
                a = atof(cell);
                isASet = 1;
            }
            else
            {
                b = atof(cell);
                isBSet = 1;
            }

            if (isASet && isBSet)
            {
                a = action(a, b, lastAction);
                b = 0;
                isBSet = 0;
            }

            memset(cell, 0, sizeof(cell));
            printf("a= %f b=%f \n", a, b);

            if (operation[i] != '\0')
                lastAction = operation[i];
        }
    }
